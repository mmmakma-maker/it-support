<script>
/**************
 * الموظفين
 **************/
const employees = [
  { name:'أحمد', dept:'network' }, { name:'منذر', dept:'network' },
  { name:'حارث', dept:'support' }, { name:'مبارك', dept:'support' }, { name:'شروق', dept:'support' }, { name:'عائشة', dept:'support' }, { name:'حيدر', dept:'support' },
  { name:'فاطمة السعدي', dept:'development' }, { name:'سهيل', dept:'development' }, { name:'فاطمة البداعي', dept:'development' }, { name:'زاهد', dept:'development' }
];

/**************
 * Auth & Logout
 **************/
const token = localStorage.getItem("token");
const userName = localStorage.getItem("user");
if (!token) window.location.href = "/login.html";
document.getElementById("createdBy").value = userName || "";

document.getElementById("logoutBtn").addEventListener("click", () => {
  localStorage.removeItem("token");
  localStorage.removeItem("user");
  window.location.href = "/login.html";
});

/**************
 * Socket.io & Notifications
 **************/
const socket = io();
const notificationSound = new Audio("/sounds/ping.mp3");

socket.on("ticketAdded", (ticket) => {
  addTicketToTable(ticket);
  notificationSound.play();
  alert(`طلب جديد تم إضافته:\nالعنوان: ${ticket.title}\nالقسم: ${ticket.department}`);
  refreshStats();
});

socket.on("ticketUpdated", (ticket) => {
  updateTicketInTable(ticket);
  notificationSound.play();
  alert(`تم تحديث طلب:\nالعنوان: ${ticket.title}\nالحالة الجديدة: ${ticket.status}`);
  refreshStats();
});

/**************
 * Tickets Form
 **************/
const form = document.getElementById("ticketForm");
form.addEventListener("submit", async (e) => {
  e.preventDefault();
  const data = {
    title: document.getElementById("title").value,
    description: document.getElementById("description").value,
    department: document.getElementById("department").value,
    createdBy: document.getElementById("createdBy").value,
    location: document.getElementById("location").value
  };
  try {
    const res = await fetch("/api/ticket", {
      method: "POST",
      headers: { "Content-Type": "application/json", "Authorization": "Bearer " + token },
      body: JSON.stringify(data)
    });
    if (!res.ok) throw new Error("فشل إرسال الطلب");
    const ticket = await res.json(); 
    addTicketToTable(ticket); 
    form.reset();
    refreshStats();
  } catch (err) { alert("خطأ: " + err.message); }
});

/**************
 * Tickets Table
 **************/
const ticketsTableBody = document.querySelector("#ticketsTable tbody");
const searchInput = document.getElementById("search");

searchInput.addEventListener("input", () => {
  const term = searchInput.value.toLowerCase();
  document.querySelectorAll("#ticketsTable tbody tr").forEach(tr => {
    const title = tr.children[0].textContent.toLowerCase();
    const department = tr.children[1].textContent.toLowerCase();
    tr.style.display = (title.includes(term) || department.includes(term)) ? "" : "none";
  });
});

function addTicketToTable(ticket) {
  const tr = document.createElement("tr");
  tr.id = ticket._id;
  tr.className = "status-" + ticket.status;
  tr.innerHTML = `
    <td>${ticket.title}</td>
    <td>${ticket.department}</td>
    <td>${ticket.description || ""}</td>
    <td>${ticket.createdBy || ""}</td>
    <td>${ticket.location || ""}</td>
    <td>${ticket.status}</td>
    <td>${ticket.assignedTo || ""}</td>
    <td>
      ${ticket.status === "open" ? `<button onclick="assignTicket('${ticket._id}')">استلام الطلب</button>` : ""}
      ${ticket.status !== "closed" ? `<button onclick="closeTicket('${ticket._id}')">إغلاق الطلب</button>` : ""}
    </td>
  `;
  ticketsTableBody.appendChild(tr);
}

function updateTicketInTable(ticket) {
  const tr = document.getElementById(ticket._id);
  if (!tr) return;
  tr.className = "status-" + ticket.status;
  tr.innerHTML = `
    <td>${ticket.title}</td>
    <td>${ticket.department}</td>
    <td>${ticket.description || ""}</td>
    <td>${ticket.createdBy || ""}</td>
    <td>${ticket.location || ""}</td>
    <td>${ticket.status}</td>
    <td>${ticket.assignedTo || ""}</td>
    <td>
      ${ticket.status === "open" ? `<button onclick="assignTicket('${ticket._id}')">استلام الطلب</button>` : ""}
      ${ticket.status !== "closed" ? `<button onclick="closeTicket('${ticket._id}')">إغلاق الطلب</button>` : ""}
    </td>
  `;
}

/**************
 * Load Tickets & Stats
 **************/
async function loadTickets() {
  try {
    const res = await fetch("/api/tickets", { headers: { "Authorization":"Bearer "+token } });
    const tickets = await res.json();
    ticketsTableBody.innerHTML = "";
    tickets.forEach(addTicketToTable);
    refreshStats();
  } catch (err) { console.error(err); }
}

function refreshStats() {
  const tickets = Array.from(ticketsTableBody.children).map(tr => ({
    assignedTo: tr.children[6].textContent,
    status: tr.children[5].textContent
  }));
  updateStats(tickets);
  renderChart(tickets);
  updateEmployeeStats(tickets);
}

/**************
 * General Stats & Chart
 **************/
function updateStats(tickets) {
  document.getElementById("totalTickets").textContent = tickets.length;
  document.getElementById("assignedTickets").textContent = tickets.filter(t => t.assignedTo).length;
  document.getElementById("openTickets").textContent = tickets.filter(t => t.status==='open').length;
}

let ticketChart;
function renderChart(tickets) {
  const ctx = document.getElementById('ticketChart').getContext('2d');
  const statusCounts = {
    open: tickets.filter(t => t.status === 'open').length,
    in_progress: tickets.filter(t => t.status === 'in_progress').length,
    closed: tickets.filter(t => t.status === 'closed').length
  };
  if(ticketChart) ticketChart.destroy();
  ticketChart = new Chart(ctx, {
    type: 'pie',
    data: {
      labels: ['مفتوحة','قيد التنفيذ','مغلقة'],
      datasets: [{
        data: [statusCounts.open,statusCounts.in_progress,statusCounts.closed],
        backgroundColor: ['#ffc107','#c4e0ff','#28a745']
      }]
    },
    options: { responsive:true, maintainAspectRatio:false, plugins:{legend:{position:'bottom'}} }
  });
}

/**************
 * Employee Stats
 **************/
function updateEmployeeStats(tickets) {
  const tableBody = document.getElementById("employeeTableBody");
  tableBody.innerHTML = "";

  const dataMap = employees.map(emp => {
    const empTickets = tickets.filter(t => t.assignedTo === emp.name);
    return {
      ...emp,
      received: empTickets.length,
      closed: empTickets.filter(t => t.status==='closed').length
    };
  });

  // جدول
  dataMap.forEach(emp => {
    const tr = document.createElement("tr");
    tr.innerHTML = `<td>${emp.name}</td><td>${emp.dept}</td><td>${emp.received}</td><td>${emp.closed}</td>`;
    tableBody.appendChild(tr);
  });

  // شارت Bar
  const ctx = document.getElementById('employeeChart').getContext('2d');
  if(window.employeeChart) window.employeeChart.destroy();
  window.employeeChart = new Chart(ctx, {
    type:'bar',
    data:{
      labels: dataMap.map(e => e.name),
      datasets:[{
        label:'الطلبات المغلقة',
        data: dataMap.map(e => e.closed),
        backgroundColor: dataMap.map(e => {
          if(e.dept==='network') return '#007BFF';
          if(e.dept==='support') return '#28a745';
          return '#ffc107';
        })
      }]
    },
    options:{
      responsive:true,
      maintainAspectRatio:false,
      plugins:{legend:{display:false}}
    }
  });
}

/**************
 * Assign & Close Tickets
 **************/
async function assignTicket(id) {
  const name = prompt("ادخل اسمك لاستلام الطلب:");
  if (!name) return;
  try {
    await fetch(`/api/ticket/${id}/assign`, {
      method:"PUT",
      headers:{ "Content-Type":"application/json", "Authorization":"Bearer "+token },
      body: JSON.stringify({ assignedTo: name })
    });
    refreshStats();
  } catch(err){ alert("خطأ: "+err.message); }
}

async function closeTicket(id) {
  try {
    await fetch(`/api/ticket/${id}/close`, {
      method:"PUT",
      headers:{ "Content-Type":"application/json", "Authorization":"Bearer "+token }
    });
    refreshStats();
  } catch(err){ alert("خطأ: "+err.message); }
}

// Start
loadTickets();
